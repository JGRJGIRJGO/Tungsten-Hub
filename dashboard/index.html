Backend Database Integration

Lyra UI Library uses Supabase (a free, open-source serverless PostgreSQL database) to enable secure user authentication, script registering, and real analytics logging directly from your GitHub Pages. Follow these quick steps to get connected:

1. Create Database Tables

Sign up at supabase.com (completely free), create a project, open the SQL Editor tab, paste this SQL script, and click Run:

-- Create Scripts Table
create table public.scripts (
  id uuid default gen_random_uuid() primary key,
  user_id uuid references auth.users(id) on delete cascade not null,
  name text not null,
  version text not null default '1.0.0',
  status text not null default 'Active',
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Create Key Settings Table
create table public.key_settings (
  script_id uuid references public.scripts(id) on delete cascade primary key,
  steps integer not null default 1,
  salt text not null default 'LyraSaltKey',
  file_name text not null default 'lyra_key.txt',
  link_1 text,
  link_2 text,
  link_3 text
);

-- Create Key Redeems Table
create table public.key_redeems (
  id uuid default gen_random_uuid() primary key,
  script_id uuid references public.scripts(id) on delete cascade not null,
  hwid text not null,
  step integer not null default 1,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);
